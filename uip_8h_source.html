1338 /* The following flags may be set in the global variable uip_flags
1339  before calling the application callback. The UIP_ACKDATA,
1340  UIP_NEWDATA, and UIP_CLOSE flags may both be set at the same time,
1341  whereas the others are mutualy exclusive. Note that these flags
1342  should *NOT* be accessed directly, but only through the uIP
1343  functions/macros. */
1344 
1345 #define UIP_ACKDATA 1 /* Signifies that the outstanding data was
1346  acked and the application should send
1347  out new data instead of retransmitting
1348  the last data. */
1349 #define UIP_NEWDATA 2 /* Flags the fact that the peer has sent
1350  us new data. */
1351 #define UIP_REXMIT 4 /* Tells the application to retransmit the
1352  data that was last sent. */
1353 #define UIP_POLL 8 /* Used for polling the application, to
1354  check if the application has data that
1355  it wants to send. */
1356 #define UIP_CLOSE 16 /* The remote host has closed the
1357  connection, thus the connection has
1358  gone away. Or the application signals
1359  that it wants to close the
1360  connection. */
1361 #define UIP_ABORT 32 /* The remote host has aborted the
1362  connection, thus the connection has
1363  gone away. Or the application signals
1364  that it wants to abort the
1365  connection. */
1366 #define UIP_CONNECTED 64 /* We have got a connection from a remote
1367  host and have set up a new connection
1368  for it, or an active connection has
1369  been successfully established. */
1370 
1371 #define UIP_TIMEDOUT 128 /* The connection has been aborted due to
1372  too many retransmissions. */
